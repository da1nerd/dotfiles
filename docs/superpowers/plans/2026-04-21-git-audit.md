# git-audit and git-grab rename implementation plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a new `bin/git-audit` sync-status reporter for repos under `~/git`, and rename the existing `bin/gitme` clone helper to `bin/git-grab` so both are usable as git subcommands.

**Architecture:** Two standalone bash scripts in `bin/` (which is on PATH). Naming as `git-*` means git treats each as a subcommand (`git grab …`, `git audit …`). `git-audit` is a single self-contained bash file — no helpers, no shared library. Read-only: never mutates any repo.

**Tech Stack:** bash, git, coreutils (`find`, `xargs`). macOS and Linux. No test harness — verification is running the script against manually-built fixtures and observing output / exit code. Matches the repo convention documented in `CLAUDE.md`.

**Spec:** `docs/superpowers/specs/2026-04-21-git-audit-design.md`

```bash
#!/usr/bin/env bash
#
# git-audit — report git repositories under [path] that have unsynced work.
#
# Usage:
#   git-audit [--fetch] [-v|--verbose] [-h|--help] [path]
#
# Default path: ~/git. Exit 0 if clean, 1 if any repos flagged, 2 on usage errors.

set -u
set -o pipefail
# Note: set -e is intentionally off. Per-repo git failures are expected and
# are handled with explicit status checks.

usage() {
  cat <<'EOF'
Usage: git-audit [--fetch] [-v|--verbose] [-h|--help] [path]

Report git repositories under [path] that are not safely synced:
uncommitted, staged, untracked, unpushed, no-upstream, or no-remote.

  --fetch        Fetch from all remotes before auditing (parallelized, 8 at a time).
  -v, --verbose  Multi-line per-repo detail instead of compact tags.
  -h, --help     Show this help and exit.

Path defaults to ~/git. Exit 0 if no repos are flagged, 1 if any are,
2 on a usage error.
EOF
}

# --- CLI parsing ---

fetch=0
verbose=0
path=""

while (( $# > 0 )); do
  case "$1" in
    -h|--help) usage; exit 0 ;;
    --fetch) fetch=1 ;;
    -v|--verbose) verbose=1 ;;
    --)
      shift
      if (( $# > 0 )); then
        if [[ -n "$path" ]]; then
          echo "git-audit: unexpected argument: $1" >&2; exit 2
        fi
        path="$1"; shift
      fi
      if (( $# > 0 )); then
        echo "git-audit: unexpected argument: $1" >&2; exit 2
      fi
      ;;
    -*)
      echo "git-audit: unknown option: $1" >&2
      usage >&2
      exit 2
      ;;
    *)
      if [[ -n "$path" ]]; then
        echo "git-audit: unexpected argument: $1" >&2; exit 2
      fi
      path="$1"
      ;;
  esac
  shift || true
done

path="${path:-$HOME/git}"

if [[ ! -d "$path" ]]; then
  echo "git-audit: not a directory: $path" >&2
  exit 2
fi

path_abs=$(cd "$path" && pwd)

# Phase B will add: discovery, per-repo checks, output formatting, fetch,
# error handling. For now the script validates args and exits clean.

exit 0
```

- new_string:
```
path_abs=$(cd "$path" && pwd)

# --- Discovery ---
#
# Walk "$path_abs" and collect every directory containing a ".git" entry
# (either a directory or a gitfile for worktrees). Stop descending once a
# repo is found, so submodules / vendored checkouts aren't audited
# separately.

repos=()
while IFS= read -r -d '' repo; do
  repos+=("$repo")
done < <(find "$path_abs" -type d \
  \( -exec sh -c 'test -e "$0/.git"' {} \; -print0 -prune \) 2>/dev/null)

# --- Per-repo checks ---
#
# For each repo, compute tags. A repo with no tags is silently skipped.
# The ordering of tags on output follows the order they're appended here.

any_flagged=0

for repo in "${repos[@]}"; do
  tags=()

  # Working-tree state from `git status --porcelain` (v1 format).
  # Columns: first = index (staged), second = worktree (unstaged).
  # "??" means untracked.
  porc=$(git -C "$repo" status --porcelain 2>/dev/null)

  if echo "$porc" | grep -q '^.[MDARCU]'; then
    tags+=("uncommitted")
  fi
  if echo "$porc" | grep -q '^[MADRCU] '; then
    tags+=("staged")
  fi
  if echo "$porc" | grep -q '^??'; then
    tags+=("untracked")
  fi

  (( ${#tags[@]} == 0 )) && continue
  any_flagged=1

  # Relative path for readable output.
  rel="${repo#$path_abs/}"
  [[ "$rel" == "$repo" ]] && rel="."

  # Compact output: path, then each tag in brackets.
  printf '%s' "$rel"
  for tag in "${tags[@]}"; do
    printf ' [%s]' "$tag"
  done
  printf '\n'
done

(( any_flagged )) && exit 1
exit 0
```

```
# Truncate a comma-joined list to the first 5 entries, append "(+N more)".
truncate_list() {
  local -a items=("$@")
  local n=${#items[@]}
  if (( n <= 5 )); then
    (IFS=", "; echo "${items[*]}")
  else
    local head=("${items[@]:0:5}")
    local extra=$(( n - 5 ))
    (IFS=", "; echo "${head[*]} (+$extra more)")
  fi
}

for repo in "${repos[@]}"; do
  tags=()
  # Per-condition detail, populated in parallel with tags for verbose output.
  detail_modified=()
  detail_untracked=()
  detail_staged=()
  detail_unpushed=""
  detail_upstream=""
  detail_branch=""

  porc=$(git -C "$repo" status --porcelain 2>/dev/null)

  # Parse porcelain lines once into per-category detail.
  while IFS= read -r line; do
    [[ -z "$line" ]] && continue
    idx="${line:0:1}"
    wt="${line:1:1}"
    rest="${line:3}"
    case "$idx$wt" in
      '??') detail_untracked+=("$rest") ;;
      *)
        if [[ "$wt" =~ [MDARCU] ]]; then
          detail_modified+=("$rest")
        fi
        if [[ "$idx" =~ [MADRCU] && "$wt" == " " ]]; then
          detail_staged+=("$rest")
        fi
        ;;
    esac
  done <<< "$porc"

  (( ${#detail_modified[@]} > 0 )) && tags+=("uncommitted")
  (( ${#detail_staged[@]} > 0 )) && tags+=("staged")
  (( ${#detail_untracked[@]} > 0 )) && tags+=("untracked")

  head_ref=$(git -C "$repo" symbolic-ref --quiet --short HEAD 2>/dev/null || true)
  remotes=$(git -C "$repo" remote 2>/dev/null)

  if [[ -z "$remotes" ]]; then
    tags+=("no-remote")
  elif [[ -n "$head_ref" ]]; then
    detail_branch="$head_ref"
    upstream=$(git -C "$repo" rev-parse --abbrev-ref --symbolic-full-name \
                 '@{u}' 2>/dev/null || true)
    if [[ -z "$upstream" ]]; then
      tags+=("no-upstream:$head_ref")
    else
      ahead=$(git -C "$repo" rev-list --count "$upstream..HEAD" 2>/dev/null || echo 0)
      if (( ahead > 0 )); then
        tags+=("unpushed:$ahead")
        detail_unpushed="$ahead"
        detail_upstream="$upstream"
      fi
    fi
  fi

  (( ${#tags[@]} == 0 )) && continue
  any_flagged=1

  rel="${repo#$path_abs/}"
  [[ "$rel" == "$repo" ]] && rel="."

  if (( verbose )); then
    printf '%s\n' "$rel"
    if (( ${#detail_modified[@]} > 0 )); then
      printf '  Modified:    %s\n' "$(truncate_list "${detail_modified[@]}")"
    fi
    if (( ${#detail_staged[@]} > 0 )); then
      printf '  Staged:      %s\n' "$(truncate_list "${detail_staged[@]}")"
    fi
    if (( ${#detail_untracked[@]} > 0 )); then
      printf '  Untracked:   %s\n' "$(truncate_list "${detail_untracked[@]}")"
    fi
    if [[ -n "$detail_unpushed" ]]; then
      printf '  Unpushed:    %s (%s commits ahead of %s)\n' \
        "$detail_branch" "$detail_unpushed" "$detail_upstream"
    fi
    # no-upstream / no-remote also deserve verbose lines.
    for tag in "${tags[@]}"; do
      case "$tag" in
        no-upstream:*) printf '  No upstream: %s\n' "${tag#no-upstream:}" ;;
        no-remote)     printf '  No remote configured\n' ;;
      esac
    done
  else
    printf '%s' "$rel"
    for tag in "${tags[@]}"; do
      printf ' [%s]' "$tag"
    done
    printf '\n'
  fi
done
```
